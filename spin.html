r < rows.length; r++) { const code = rows[r][1]; const time = rows[r][8]; // column I if (!code || !time) continue; try { const rowDay = Utilities.formatDate(new Date(time), tz, "yyyy-MM-dd"); if (rowDay === todayKey) usedToday.add(String(code));
